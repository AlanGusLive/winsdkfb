194  String^ FacebookDialogName
195  )
196 {
197  FBSession^ sess = FBSession::ActiveSession;
198 
199  //
200  // This looks strange, but is correct. One side or the other of this

252 Uri^ FacebookDialog::BuildLoginDialogUrl(
253  PropertySet^ Parameters
254  )
255 {
256  FBSession^ s = FBSession::ActiveSession;
257  String^ apiVersion = L"";
258  if (s->APIMajorVersion)
259  {
260  apiVersion = L"v" + s->APIMajorVersion.ToString() + L"." + s->APIMinorVersion.ToString() + L"/";
261  }
262  String^ uriString = L"https://" + GetFBServer() +
263  L".facebook.com/" + apiVersion + L"dialog/oauth?client_id=" + s->FBAppId;
264 
265  // Use some reasonable default login parameters
266  String^ scope = DefaultScope;
267  String^ displayType = DefaultDisplay;
268  String^ responseType = DefaultResponse;
269 
270  uriString += L"&redirect_uri=" + GetRedirectUriString(L"login");
271 
272  // Enumerate through all the parameters
273  IIterator<IKeyValuePair<String^, Object^>^>^ first = Parameters->First();
274  while (first && (first->HasCurrent))
275  {
276  String^ Key = first->Current->Key;
277  String^ Value = dynamic_cast<String^>(first->Current->Value);
278  if (Value)
279  {
280  if (!String::CompareOrdinal(Key, ScopeKey))
281  {
282  scope = Value;
283  }
284  else if (!String::CompareOrdinal(Key, DisplayKey))
285  {
286  displayType = Value;
287  }
288  else if (!String::CompareOrdinal(Key, ResponseTypeKey))
289  {
290  responseType = Value;
291  }
292  else
293  {
294  uriString += L"&" + Key + L"=" + Value;
295  }
296  }
297 
298  first->MoveNext();
299  }
300 
301  uriString += L"&" + ScopeKey + L"=" + scope + L"&" + DisplayKey + L"=" +
302  displayType + L"&" + ResponseTypeKey + L"=" + responseType;
303 
304  return ref new Uri(uriString);
305 }
