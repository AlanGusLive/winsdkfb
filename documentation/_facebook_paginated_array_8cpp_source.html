212 FBResult^ FBPaginatedArray::ConsumePagedResponse(
213  String^ JsonText
214  )
215 {
216  FBResult^ result = nullptr;
217  JsonValue^ Value = nullptr;
218  bool foundPaging = false;
219  bool foundData = false;
220  bool foundError = false;
221 
222  if (JsonValue::TryParse(JsonText, &Value))
223  {
224  if (Value->ValueType == JsonValueType::Object)
225  {
226  JsonObject^ obj = Value->GetObject();
227  IIterator<IKeyValuePair<String^, IJsonValue^>^>^ it = nullptr;
228  for (it = obj->First(); it->HasCurrent; it->MoveNext())
229  {
230  if (!String::CompareOrdinal(it->Current->Key, L"error"))
231  {
232  FBError^ err = FBError::FromJson(
233  it->Current->Value->Stringify());
234  result = ref new FBResult(err);
235  foundError = true;
236  break;
237  }
238  else if (!String::CompareOrdinal(it->Current->Key, L"paging"))
239  {
240  _paging = static_cast<FBPaging^>(
241  FBPaging::FromJson(it->Current->Value->Stringify()));
242  if (_paging)
243  {
244  foundPaging = true;
245  }
246  }
247  else if (!String::CompareOrdinal(it->Current->Key, L"data"))
248  {
249  if (it->Current->Value->ValueType != JsonValueType::Array)
250  {
251  throw ref new InvalidArgumentException(
252  SDKMessageBadObject);
253  }
254 
255  _current = ObjectArrayFromJsonArray(
256  it->Current->Value->GetArray(), _objectFactory);
257  if (_current)
258  {
259  foundData = true;
260  }
261  }
262  }
263  }
264 
265  // If all data for the array fits in one page, we don't get a 'paging'
266  // object. We should always get a data object, though. Note that a FB
267  // error should not result in a thrown exception!
268  if (!(foundError || foundData))
269  {
270  throw ref new InvalidArgumentException(SDKMessageBadObject);
271  }
272 
273  if (!foundError)
274  {
275  result = ref new FBResult(_current);
276  }
277  }
278 
279  return result;
280 }
